\section{Introduction and Problem Statement}

\begin{frame}{The "Black Box" of Sleep Analysis}
    \begin{itemize}
        \item \textbf{Historical Context:} Clinical sleep analysis has relied on the manual or semi-automated visual scoring of polysomnograms.
        \item \textbf{The AASM Standard:} Limits continuous neural activity into discrete, rigid sleep stages (Wake, REM, N1, N2, N3).
        \item \textbf{The Flaw:} This macroscopic staging paradigm inherently filters out the high-dimensional spatial and temporal variability intrinsic to internal cognitive phenomena.
    \end{itemize}
\end{frame}

\begin{frame}{The Cortical Hot Zone}
    \begin{itemize}
        \item \textbf{Localization:} Posterior parahippocampal and occipital regions are strongly correlated with dream onset and conscious recall (Siclari et al., 2018).
        \item \textbf{Requirements:} Investigating this zone requires continuous, high-fidelity signal analysis exceeding standard 30-second epoch scoring.
        \item \textbf{The Challenge:} How do we map these dense temporal structures to subjective dream contents?
    \end{itemize}
\end{frame}

\begin{frame}{The Data Scarcity Problem}
    \begin{itemize}
        \item \textbf{Subjective Bottleneck:} Biological subjects can only report subjective conscious experiences upon awakening.
        \item \textbf{Imbalanced Targets:} Standard supervised classification models are forced to overfit against highly imbalanced baselines (mostly Neutral/NREM, very few "Dream" reports).
        \item \textbf{Gradient Distortion:} Rare cognitive events lack the scale required to train modern deep neural networks.
    \end{itemize}
\end{frame}

\begin{frame}{Why Existing Generative AI Fails}
    \begin{itemize}
        \item \textbf{Spectral Collapse:} Attempting to augment scarce REM/NREM data with unconditional GANs or basic Diffusion models results in topological failure.
        \item \textbf{The $1/f$ Rule:} Synthetic tensors generated by unconstrained AI fail to respect the physiological $1/f$ energy scaling laws of the mammalian cortex.
        \item \textbf{Temporal Decay:} Sequential patterns dissolve over extended epochs ($>1000$ sampling points).
    \end{itemize}
\end{frame}

\begin{frame}{The Generation-Interpretation Duality}
    \begin{block}{The Core BCI Gap}
        Existing algorithms suffer from computational decoupling. They can either generate raw physiological signals (without knowing what the signals mean) OR they can classify existing signals (but cannot synthesize new ones from text).
    \end{block}
    \begin{itemize}
        \item How do we map macroscopic physical EEG geometry to abstract human language?
    \end{itemize}
\end{frame}

\begin{frame}{The Proposed Solution: AC-Semantic-TimeGAN}
    \begin{itemize}
        \item \textbf{Phase 1 (Physical):} A physiology-aware foundational generation framework (AC-TimeGAN) integrating a Temporal Supervisor, an active Auxiliary Classifier, and Dynamic Mode Decomposition (DMD).
        \item \textbf{Phase 2 (Semantic):} A zero-shot cross-modal extension incorporating an InfoNCE projection head to align stable temporal outputs with NLP transformers.
    \end{itemize}

        \includegraphics[width=0.55\textwidth]{results/system_arch.png}
    \end{center}
\end{frame}

\section{Phase 1: Physiology-Aware Generation}

\begin{frame}{Architecture: AC-TimeGAN Mechanics}
    \begin{itemize}
        \item \textbf{Noise Variance:} EEG arrays contain extreme noise. Running adversarial learning on raw $\mathbb{R}^{T \times C}$ disrupts temporal gradients.
        \item \textbf{Gated Temporal Embedding:} We execute auto-encoding geometry using Gated Recurrent Units (GRUs):
        \begin{equation}
            \mathbf{h}_t = (1 - \mathbf{z}_t) \odot \mathbf{h}_{t-1} + \mathbf{z}_t \odot \tilde{\mathbf{h}}_t
        \end{equation}
        \item Shrinks raw dimensionality into an invariant sequence manifold.
    \end{itemize}
\end{frame}

\begin{frame}{Overcoming Sequence Decay (Temporal Supervision)}
    \begin{itemize}
        \item A separate \textbf{Supervisor Network ($\mathcal{S}$)} observes generated temporal steps.
        \item Penalizes forward predictions that violate physical flow probability:
        \begin{equation}
            \mathcal{L}_{sup} = \mathbb{E}_{\mathbf{X}, Y} \left[ || \mathbf{h}_t - \mathcal{S}(\mathbf{h}_{t-1}) ||_2^2 \right]
        \end{equation}
        \item Prevents gradient explosion during long temporal rollouts.
    \end{itemize}
\end{frame}

\begin{frame}{Fighting Dataset Imbalance (Auxiliary Classifier)}
    \begin{itemize}
        \item Standard discriminators perform strictly binary classification (Real vs Synthesized).
        \item We split the discriminator to include an active \textbf{Auxiliary Classifier ($\mathcal{C}$)}.
        \item $\mathcal{C}$ concurrently maps conditional targets (e.g., distinguishing Dream states from Neutral baselines), forcing distinct vector separations.
    \end{itemize}
\end{frame}

\begin{frame}{The Flaw in GANs: Spectral Drift}
    \begin{itemize}
        \item Generative variance forces synthetic vectors off biological constraints over time.
        \item Deep neural architectures natively map to Euclidean space, but brainwaves are organized across strictly bounded oscillatory frequencies (Delta $>$ Theta $>$ Alpha $>$ Beta).
        \item We require a physics-based governor.
    \end{itemize}
\end{frame}

\begin{frame}{Enter Fluid Dynamics: DMD}
    \begin{itemize}
        \item \textbf{Dynamic Mode Decomposition (DMD):} Algorithmically extracted absolute complex eigen-frequency bounding thresholds ($\omega_{real}$) governing physical mammalian brain states.
        \item Singular Value Decomposition (SVD) matrix extracts true coefficients:
        \begin{equation}
            \tilde{A} = U^* Y V \Sigma^{-1} \ \Rightarrow \ \omega_{real} = \frac{\ln(\lambda)}{\Delta t}
        \end{equation}
    \end{itemize}
\end{frame}

\begin{frame}{Physiology-Aware Alignment (PAA) Penalty}
    \begin{itemize}
        \item When the network generates sequences resulting in Spectral Collapse, we do not discard them.
        \item We enforce a deterministic banded adjustment to project the anomalous waveform precisely onto the continuous physiological $1/f$ curve:
        \begin{equation}
            \hat{X}_{aligned} (f) = \mathcal{F}(\hat{X}_{proj}) \cdot \sqrt{\frac{\sigma^2_{real}(f) + \epsilon}{\sigma^2_{synth}(f) + \epsilon}}
        \end{equation}
    \end{itemize}
\end{frame}

\begin{frame}{Algorithm Execution: DMD-Bounded Synthesis}
    \textbf{Algorithm 1 execution loop:}
    \begin{enumerate}
        \item Embed coordinate matrices ($\mathbf{h} = e(\mathbf{x})$)
        \item Compute unconditional noise projection
        \item Evaluate Auxiliary Classification and Jensen-Shannon entropy
        \item Extract FFT scale and apply PAA continuous scaling penalty gradient
        \item Backpropagate consolidated geometry
    \end{enumerate}
\end{frame}

\begin{frame}{Phase 1 Sub-Conclusion}
    \begin{center}
        \Large By penalizing generator drift via eigen-frequency boundary curves, we have successfully developed an AI block mathematically restricted to synthesizing structurally valid human brainwaves.
    \end{center}
\end{frame}

\section{Phase 2: Semantic Text Alignment}

\begin{frame}{The Cross-Modal Challenge}
    \begin{itemize}
        \item We possess mathematically perfect, synthesized brainwaves.
        \item How do we map these $\mathbb{R}^{T \times C}$ physical tensors directly onto raw English words describing affective emotions?
        \item \textbf{Target:} True Natural Language Processing (NLP) integration into physiological manifolds.
    \end{itemize}
\end{frame}

\begin{frame}{Freezing the Physical Map}
    \begin{itemize}
        \item If we execute end-to-end multi-modal optimization, the fragile $1/f$ physiological boundary collapses.
        \item \textbf{Solution:} Phase 2 executes entirely with Phase 1 Generator and Encoder weights \textbf{frozen}.
        \item We only adjust the cross-modal projection logic, anchoring semantics directly to the invariant physiological structure.
    \end{itemize}
\end{frame}

\begin{frame}{NLP Integration via Transformers}
    \begin{itemize}
        \item Natural language processing utilizes a pre-trained contextual Transformer (distilled MiniLM-L6).
        \item Raw subjective dream paragraphs are translated into a dense hyperspherical array.
        \item Yields fixed semantic variables: $\mathcal{Y}_{emb} \in \mathbb{R}^{D=384}$.
    \end{itemize}
\end{frame}

\begin{frame}{InfoNCE Contrastive Geometry}
    \begin{itemize}
        \item Phase 2 introduces a contrastive nonlinear projection head ($P_{sim}$).
        \item It warps physiological temporal sequences $\mathbf{h}_i$ to minimize their geometric distance against the corresponding linguistic anchor vector $\mathbf{y}_i$:
        \begin{equation}
            \mathcal{L}_{InfoNCE} = - \mathbb{E} \left[ \log \frac{\exp(\text{sim}(P_s(\mathbf{h}_i), \mathbf{y}_i) / \tau)}{\sum \exp(\text{sim}(P_s(\mathbf{h}_i), \mathbf{y}_j) / \tau)} \right]
        \end{equation}
    \end{itemize}
\end{frame}

\begin{frame}{The Dual-Modality Target}
    \begin{block}{Text-to-Brainwave}
        The InfoNCE equation penalizes decoupled vectors by expanding the denominator. By actively enforcing this loss, the architecture translates an abstract english prompt into a perfectly bounded physical output.
    \end{block}
\end{frame}

\begin{frame}{The Zero-Shot Setup}
    \begin{itemize}
        \item Validating true comprehension requires \textbf{Zero-Shot} execution.
        \item Textual anchors remain strictly isolated across training boundaries to prevent cross-modal semantic leakage.
        \item Evaluates if the system maps generalized linguistic meaning to generalized temporal bounds, rather than just memorizing subject data.
    \end{itemize}
\end{frame}

\begin{frame}{The Temperature Scaling Factor ($\tau$)}
    \begin{itemize}
        \item Empirically optimized via matrix grid search across $\tau \in [0.01, 0.1]$.
        \item Locked at standard $\tau=0.07$.
        \item Maximizes textual cluster separation inside the hypersphere without collapsing the nonlinear projection.
    \end{itemize}
\end{frame}

\begin{frame}{Phase 2 Sub-Conclusion}
    \begin{center}
        \Large Biological sequence outputs are no longer blind variables. They are mathematically linked directly to human-comprehensible semantic definitions.
    \end{center}
\end{frame}

\section{Datasets and Verification Protocols}

\begin{frame}{The Datasets}
    \begin{itemize}
        \item \textbf{DREAM Database (Wong et al.):} 64-channel arrays mapped to complex Free Text subjective paragraphs. Used for initializing the dense text vectorization.
        \item \textbf{DEED Database (Zheng et al.):} 32-channel arrays spanning 533 sequences rigidly tagged to discrete affective milestones. 
        \item All networks downsampled uniformly to a 32-channel 10-20 international topography.
    \end{itemize}
\end{frame}

\begin{frame}{Rigorous Structural Evaluation}
    \begin{itemize}
        \item \textbf{TSTR (Train-Synthetic, Test-Real):} We train an entirely separate, uninformed Random Forest or 1D-CNN classifier exclusively on the outputs generated by the AI.
        \item If it successfully identifies real-world neurological data afterwards, the synthetic structure shares an identical mathematical manifold with biology.
    \end{itemize}
\end{frame}

\begin{frame}{Isolating the Semantics (LOSO)}
    \begin{itemize}
        \item Prevent morphological memorization utilizing \textbf{Leave-One-Subject-Out (LOSO)} cross-validation.
        \item Each linguistic dream report is utilized exactly once. 
        \item Cross-subject text embedding reuse is strictly eliminated.
    \end{itemize}
\end{frame}

\begin{frame}{Hardware & Statistical Checks}
    \begin{itemize}
        \item Executed natively natively on a tensor-optimized 24GB local processing architecture.
        \item All quantitative metrics represent the mean results averaged across $N=5$ randomly seeded initialization runs.
        \item Statistical significance ($p < 0.05$) explicitly evaluated using permutation testing (10,000 iterations).
    \end{itemize}
\end{frame}

\begin{frame}{Precision vs Recall Metrics}
    \begin{itemize}
        \item Phase 1 Evaluates: Precision, Recall, Macro F1, and Weighted Accuracy of TSTR classifiers.
        \item Phase 2 Evaluates: Top-1 Retrieval (Hits@1) and Semantic Neighborhood Retrieval (Hits@3) linking biological inputs to exact linguistic targets.
    \end{itemize}
\end{frame}

\begin{frame}{Key Diagnostic Limitations}
    \begin{itemize}
        \item \textbf{High-Frequency Deterioration:} GRUs naturally bias to low-frequency Delta state arcs; Gamma ($>30$Hz) prediction remains difficult.
        \item \textbf{Alpha-Band Constraints:} Variable language parameters trigger slight center-frequency drift in localized generation boundaries.
    \end{itemize}
\end{frame}

\begin{frame}{Computational Hardware Diagnostics}
    \begin{itemize}
        \item $\mathcal{O}(n^3)$ mathematical complexity currently blocks ambulatory processing limits.
        \item Dense Dynamic Mode SVD computation restricts Phase 1 evaluation exclusively to high-performance GPU tensor clusters rather than real-time BCI wearable caps.
    \end{itemize}
\end{frame}
